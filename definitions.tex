            % Format the table
            colspec = {ccc},
            cell{\condensedRow}{\confinedColumns} = {\condensedRowColour},
            cell{\confinedRows}{\condensedColumn} = {\condensedColumnColour},
            cell{\confinedRows}{\confinedColumns} = {\confinedColour},
            vlines = {1-3}{solid},
            vline{1, 4} = {0pt},
            hlines = {1-3}{solid},
